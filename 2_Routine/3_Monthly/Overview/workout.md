			if(workout){
			data[date] += habits[habit].replace('{workout}', workout) + '\n';
			}
			link[date] +=

		}
	}
}

let calendarData = []
for (let date in data) {
	calendarData.push({date: date, content: data[date], link: link[date]})
}
